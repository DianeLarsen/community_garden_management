to start postgresql
# Check the status of PostgreSQL
sudo /etc/init.d/postgresql status

# Start PostgreSQL
sudo /etc/init.d/postgresql start

# Stop PostgreSQL
sudo /etc/init.d/postgresql stop

# Restart PostgreSQL
sudo /etc/init.d/postgresql restart
